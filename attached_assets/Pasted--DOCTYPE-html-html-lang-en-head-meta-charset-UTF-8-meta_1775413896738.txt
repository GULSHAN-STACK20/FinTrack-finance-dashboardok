  function animateNumber(id, target, currency) {
    const el = document.getElementById(id);
    const start = 0, dur = 900;
    const startTime = performance.now();
    function step(now) {
      const p = Math.min((now - startTime) / dur, 1);
      const eased = 1 - Math.pow(1 - p, 3);
      const val = Math.round(start + (target - start) * eased);
      el.textContent = currency ? fmtFull(val) : val;
      if (p < 1) requestAnimationFrame(step);
    }
    requestAnimationFrame(step);
  }

  // ── TRANSACTIONS ──────────────────────────────────────────────────────────
  function populateFilters() {
    const cats = [...new Set(state.transactions.map(t => t.category))].sort();
    const catSel = document.getElementById('filter-category');
    catSel.innerHTML = '<option value="">All Categories</option>' + cats.map(c => `<option value="${c}">${c}</option>`).join('');

    const months = [...new Set(state.transactions.map(t => t.date.substring(0,7)))].sort().reverse();
    const mSel = document.getElementById('filter-month');
    mSel.innerHTML = '<option value="">All Months</option>' + months.map(m => `<option value="${m}">${getMonthLabel(m+'-01')}</option>`).join('');
  }

  function getFilteredTransactions() {
    return state.transactions
      .filter(t => {
        if (state.filterType && t.type !== state.filterType) return false;
        if (state.filterCat && t.category !== state.filterCat) return false;
        if (state.filterMonth && !t.date.startsWith(state.filterMonth)) return false;
        if (state.searchQuery) {
          const q = state.searchQuery.toLowerCase();
          if (!t.desc.toLowerCase().includes(q) && !t.category.toLowerCase().includes(q)) return false;
        }
        return true;
      })
      .sort((a, b) => {
        const dir = state.sortDir === 'asc' ? 1 : -1;
        if (state.sortKey === 'amount') return (a.amount - b.amount) * dir;
        if (state.sortKey === 'date') return a.date.localeCompare(b.date) * dir;
        return a[state.sortKey].localeCompare(b[state.sortKey]) * dir;
      });
  }

  function applyFilters() {
    state.searchQuery = document.getElementById('search-input').value.trim();
    state.filterType = document.getElementById('filter-type').value;
    state.filterCat = document.getElementById('filter-category').value;
    state.filterMonth = document.getElementById('filter-month').value;
    renderTransactions();
  }

  function sortBy(key) {
    if (state.sortKey === key) state.sortDir = state.sortDir === 'asc' ? 'desc' : 'asc';
    else { state.sortKey = key; state.sortDir = 'desc'; }
    document.querySelectorAll('.tx-table th').forEach(th => th.classList.remove('sorted'));
    const th = document.getElementById('th-' + key);
    if (th) th.classList.add('sorted');
    renderTransactions();
  }

  function renderTransactions() {
    const data = getFilteredTransactions();
    document.getElementById('tx-count').textContent = `${data.length} of ${state.transactions.length} transactions`;

            <span class="type-pill type-${tx.type}">

  // ── MODAL ─────────────────────────────────────────────────────────────────
  function openModal(editId = null) {
    if (state.role !== 'admin') { showToast('Viewers cannot add transactions', 'error'); return; }

    const overlay = document.getElementById('modal-overlay');
    document.getElementById('modal-edit-id').value = editId || '';
    document.getElementById('modal-desc').value = '';
    document.getElementById('modal-amount').value = '';
    document.getElementById('modal-date').value = new Date().toISOString().split('T')[0];
    document.getElementById('modal-type').value = 'expense';
    updateCategoryOptions();

    if (editId) {
      const tx = state.transactions.find(t => t.id === editId);
      if (tx) {
        document.getElementById('modal-title').textContent = 'Edit Transaction';
        document.getElementById('modal-sub').textContent = 'Update the transaction details below';
        document.getElementById('modal-desc').value = tx.desc;
        document.getElementById('modal-amount').value = tx.amount;
        document.getElementById('modal-date').value = tx.date;
        document.getElementById('modal-type').value = tx.type;
        updateCategoryOptions();
        document.getElementById('modal-category').value = tx.category;
      }
    } else {
      document.getElementById('modal-title').textContent = 'Add Transaction';
      document.getElementById('modal-sub').textContent = 'Record a new income or expense';
    }

    overlay.classList.add('open');
    document.getElementById('modal-desc').focus();
  }

  function closeModal() { document.getElementById('modal-overlay').classList.remove('open'); }

  function updateCategoryOptions() {
    const type = document.getElementById('modal-type').value;
    const cats = type === 'income' ? INCOME_CATS : EXPENSE_CATS;
    const sel = document.getElementById('modal-category');
    sel.innerHTML = cats.map(c => `<option value="${c}">${CAT_ICONS[c]||''} ${c}</option>`).join('');
  }

  function saveTransaction() {
    const editId = document.getElementById('modal-edit-id').value;
    const desc = document.getElementById('modal-desc').value.trim();
    const amount = parseFloat(document.getElementById('modal-amount').value);
    const date = document.getElementById('modal-date').value;
    const type = document.getElementById('modal-type').value;
    const category = document.getElementById('modal-category').value;

    if (!desc) { showToast('Please enter a description', 'error'); return; }
    if (!amount || amount <= 0) { showToast('Please enter a valid amount', 'error'); return; }
    if (!date) { showToast('Please pick a date', 'error'); return; }

    if (editId) {
      const idx = state.transactions.findIndex(t => t.id === parseInt(editId));
      if (idx !== -1) state.transactions[idx] = { ...state.transactions[idx], desc, amount, date, type, category };
      showToast('Transaction updated!', 'success');
    } else {
      state.transactions.unshift({ id: Date.now(), date, desc, category, type, amount });
      showToast('Transaction added!', 'success');
    }

    saveState(); closeModal(); refreshAll();
  }

  function editTx(id) { openModal(id); }

  function deleteTx(id) {
    if (!confirm('Delete this transaction?')) return;
    state.transactions = state.transactions.filter(t => t.id !== id);
    saveState(); refreshAll();
    showToast('Transaction deleted', 'success');
  }

    document.getElementById('tab-' + tab).classList.add('active');

  function exportCSV() {
    const rows = [['Date','Description','Category','Type','Amount']];
    state.transactions.forEach(t => rows.push([t.date, t.desc, t.category, t.type, t.amount]));
    const csv = rows.map(r => r.join(',')).join('\n');
    const a = document.createElement('a');
    a.href = 'data:text/csv;charset=utf-8,' + encodeURIComponent(csv);
    a.download = 'fintrack-export.csv';
    a.click();
    showToast('CSV exported!', 'success');
  }

  // ── TOAST ─────────────────────────────────────────────────────────────────

  function showToast(msg, type = 'success') {
    const toast = document.getElementById('toast');

    icon.className = type === 'success' ? 'fas fa-check-circle' : 'fas fa-exclamation-circle';
